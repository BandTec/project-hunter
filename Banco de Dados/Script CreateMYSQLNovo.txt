Create table gamer(
idGamer int primary key AUTO_INCREMENT,
nome varchar (60),
cpf char (30),
email varchar (60),
senha varchar (20),
telefone char (11)
);

Create table endereco(
idEndereco int primary key AUTO_INCREMENT,
logradouro varchar (60),
numero int,
complemento varchar (20),
bairro varchar (30),
cidade varchar (50),
uf char (2),
cep int,
fkGamer int references gamer (idGamer)
);

Create table jogo(
idJogo int primary key AUTO_INCREMENT,
nomeJogo varchar (100),
qtdPlayers int (3)
);

Create table equipe (
idEquipe int primary key AUTO_INCREMENT,
nome_equipe varchar(20),
pais varchar(30),
regiao varchar(10)
);

Create table tipoInfracao (
idTipoInfracao int primary key AUTO_INCREMENT,
tipoInfracao varchar (50)
);


create table partida (
idPartida int primary key AUTO_INCREMENT,
fkEquipe int references equipe (idEquipe),
fkJogo int references jogo (idJogo),
fkGamer int references gamer (idGamer)
);
